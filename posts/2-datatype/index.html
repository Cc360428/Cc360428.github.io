数据类型

  • 整型：在 Golang 中，整型分为有符号整型和无符号整型。有符号整型的取值范围为 -2^31 到 2^31-1，无符号整型的取值范围为 0 到 2^32-1。常见的整型类型

    • int8: 有符号 8 位整数，取值范围为 -128 到 127
    • int16: 有符号 16 位整数，取值范围为 -32768 到 32767
    • int32: 有符号 32 位整数，取值范围为 -2147483648 到 2147483647
    • int64: 有符号 64 位整数，取值范围为 -9223372036854775808 到 9223372036854775807
    • uint8: 无符号 8 位整数，取值范围为 0 到 255
    • uint16: 无符号 16 位整数，取值范围为 0 到 65535
    • uint32: 无符号 32 位整数，取值范围为 0 到 4294967295
    • uint64: 无符号 64 位整数，取值范围为 0 到 18446744073709551615

  • 浮点型：在 Golang 中，浮点型分为 float32 和 float64 两种类型。float32 占用 4 个字节，float64 占用 8 个字节，分别对应单精度浮点数和双精度浮点数

    • float32: 单精度浮点数，精度为 7 位小数
    • float64: 双精度浮点数，精度为 15 位小数
  • 布尔型：布尔型只有两个取值：true 和 false。在 Golang 中，布尔型使用 bool 类型表示

  • 字符串型：字符串是一组字符的序列，Golang 中的字符串使用 string 类型表示

    1

        var str string = "hello world"

    在 Golang 中，字符串是不可变的，如果需要修改字符串的内容，需要将其转换成 byte 类型或者 rune 类型进行修改

  • 指针：指针是一个变量，它存储了另一个变量的内存地址。在 Golang 中，指针使用 * 符号表示

    1

        var ptr *int

    这里定义了一个名为 ptr 的指向 int 类型变量的指针。如果需要获取指针所指向的变量的值，需要使用 *ptr

    1
2
3

        var x int = 10
    var ptr *int = &x
    fmt.Println(*ptr) // 输出 10

注意事项

iota 表示从 0 开始自动加 1，所以 i=1«0, j=3«1（« 表示左移的意思），即：i=1, j=6，这没问题，关键在 k 和 l，从输出结果看 k=3«2，l=3«3。

简单表述:
  • i=1
  • ：左移 0 位,不变仍为 1;
  • j=3：左移 1 位,变为二进制 110, 即 6;
  • k=3：左移 2 位,变为二进制 1100, 即 12;
  • l=3：左移 3 位,变为二进制 11000,即 24。

package main

// 要使用的包在 import
import (
	"errors"
	"fmt"
	"github.com/Cc360428/HelpPackage/UtilsHelp/logs"
)

func main() {
	str := "可以使用  := 快速定义类型  := 后面是什么类型就是什么类型"
	fmt.Println(str)
	var b bool
	fmt.Println(b)
	var age int
	age = 18
	logs.Info(age)
	var name string
	name = "lcc" + "Hyper" + "Cc"
	logs.Info(name)
	logs.Info(name, age)
	var p float64
	p = 1.11*22/2 + 100 - 112.21
	logs.Info(p)
	var a, d string
	a = "a"
	d = "d"
	logs.Info(a, d)
	var a1, a2 = 12, 22
	logs.Info(a1, a2)
	s1, s2 := "lcc", "ok"
	logs.Info(s1, s2)
	var f float64
	logs.Info(f)
	var s *string
	logs.Info(s)
	var err error
	err = errors.New("error 为错误类型")
	if err != nil {
		logs.Error(err.Error())
	}
	var array [2]int
	array[0] = 2
	array[1] = 2323
	res := fmt.Sprintf("长度%d,容量%d", len(array), cap(array))
	logs.Info(res, "内容", array, "可以通过下标查找更改", array[1])
	var chanType chan int
	logs.Info(chanType)
	mapType := make(map[string]interface{})
	mapType["name"] = "lcc"
	mapType["age"] = "18"
	logs.Info(mapType)
	type User struct {
		Name string
		Age  int
	}
	user := new(User)
	user.Name = "Cc"
	user.Age = 23
	logs.Info(user.Age, user.Name)
	logs.Info(user)
	// 常量
	const LENGTH int = 10
	const WIDTH int = 5
	var area int
	area = LENGTH * WIDTH
	logs.Info("面积为 : %d ,可以做运算", area)
	const (
		aa1 = iota
		aa2
		aa3
	)
	logs.Info(aa1, aa2, aa3)
	const (
		a11 = iota // 0
		b12        // 1
		c13        // 2
		d14 = "ha" // 独立值，iota += 1
		e15        // "ha"   iota += 1
		f16 = 100  // iota +=1
		g17        // 100  iota +=1
		h18 = iota // 7,恢复计数
		i19        // 8
	)
	logs.Info(a11, b12, c13, d14, e15, f16, g17, h18, i19)
	const (
		i = 1 << iota
		j = 3 << iota
		k
		l
	)
	logs.Info(i, j, k, l)

}
