      void straightsort( list r );
{                                               /* з”ЁзӣҙжҺҘжҸ’е…ҘжҺ’еәҸжі•еҜ№r[1]вҖҰr[n]иҝӣиЎҢжҺ’еәҸ */
  for ( i = 2; i <= n; i++ )                    /* nдёәиЎЁй•ҝпјҢд»Һз¬¬дәҢдёӘи®°еҪ•иҝӣиЎҢжҸ’е…Ҙ */
  {
    r[0] = r[i];                         /* з¬¬iдёӘи®°еҪ•еӨҚеҲ¶дёәеІ—е“Ё */
    j = i - 1;
    while ( r[0].key < r[j].key )            /* дёҺеІ—е“ЁжҜ”иҫғпјҢзӣҙеҲ°еҒҘдёҚеӨ§дәҺеІ—е“Ёй”®еҖј */
    {
      r[j + 1] = r[j];                 /* е°Ҷз¬¬jдёӘи®°еҪ•иөӢеҖјз»ҷз¬¬j+1дёӘи®°еҪ• */
      j  = j - 1;
    }
    r[j + 1] = r[0];                       /* е°Ҷз¬¬iдёӘи®°еҪ•жҸ’е…ҘеҲ°еәҸеҲ—дёӯ */
  }
}

  {

    for ( j = 1; j <= n - i; j++ )

        p  = r[j]; r[j] = r[j + 1];
        r[j + 1] = p;
      }
    if ( flag )
      return;
  }
}

    void  quickpass( list r, int h, int p )
{
  /* еҜ№йЎәеәҸиЎЁrдёӯзҡ„еӯҗеәҸеҲ—r[h]иҮіr[p]иҝӣиЎҢеҝ«йҖҹжҺ’еәҸ */
  i = h;    j = p;  /* е·ҰеҸіжҢҮй’ҲзҪ®еҲқеҖј */
  x = r[h];         /* еҸ–еӨ„зҗҶе…ғзҙ пјҲеҚідҪңдёәжһўиҪҙи®°еҪ•пјү */
  while ( i пјң j  )       /* е·ҰеҸіжҢҮй’ҲжңӘзў°еӨҙеҲҷеҸҚеӨҚеҒҡпјҡ */
  {
    while ( r[j].key > x.key && i < j )
      --j;
    /*еҸіиҫ№жңӘжүҫеҲ°е°Ҹе…ій”®еӯ—пјҢеҲҷеҸіжҢҮй’Ҳjз»§з»ӯе·Ұз§»*/
    if ( i < j )    /*еҸіиҫ№жүҫеҲ°жҜ”жһўиҪҙи®°еҪ•е°Ҹзҡ„и®°еҪ•пјҢеҲҷе°Ҷе…¶йҖҒеҲ°е·Ұиҫ№*/
    {
      r[i] = r[j];
      ++i;
    }
    while ( r[i].key <= x.key && i < j )
      ++i;
    /*иҫ№жңӘжүҫеҲ°еӨ§е…ій”®еӯ—пјҢеҲҷе·ҰжҢҮй’Ҳiз»§з»ӯеҸіз§»*/
    if ( i < j )                    /*е·Ұиҫ№жүҫеҲ°жҜ”жһўиҪҙи®°еҪ•еӨ§зҡ„и®°еҪ•пјҢеҲҷе°Ҷе…¶йҖҒеҲ°еҸіиҫ№*/
    {
      r[j] = r[i];
      - -j;
    }
  }
  r[i] = x;                               /*жһўиҪҙи®°еҪ•е®ҡдҪҚ*/
  if ( h < i - 1 )
    quickpass( r, h, i - 1 );       /*еҜ№е·ҰеӯҗеәҸеҲ—иҝӣиЎҢеҝ«йҖҹжҺ’еәҸ*/
  if ( j + 1 < high )
    quickpass( r, j + 1, p );       /*еҜ№еҸіеӯҗеәҸеҲ—иҝӣиЎҢеҝ«йҖҹжҺ’еәҸ*/
}
