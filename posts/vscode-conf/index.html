vscode 配置

{
    "version": "该配置文件的版本，由 VSCode 自动生成",
    "configurations": [
        {
            "name": "名字，好听就行",
            "type": "语言类型，装了插件后填 go 就会调用 dlv 命令",
            "request": "launch 是启动新进程，append 是附加到现有的进程",
            "mode": "dlv 的一些选项，支持 debug、exec、test、remote",
            "host": "dlv 监听的地址",
            "port": "dlv 监听的端口（此处应该是数字）",
            "program": "本地的程序路径",
            "cwd": "当前工作路径",
            "env": {
                "环境变量名": "环境变量值"
            },
            "args": ["程序启动参数"],
            "showLog": "是否输出 dlv 的日志（此处应该是 true/false）"
        }
    ]
}

            "showLog": true

    ]
