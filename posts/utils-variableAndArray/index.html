package main

import (
	"fmt"
	"strconv"
)

func main() {
	var str string
	fmt.Printf("str type:%T\n", str)
	str = "123"
	var strInt int
	// string > int
	strInt, _ = strconv.Atoi(str)
	fmt.Println(str, strInt)
	var strInt64 int64
	// string > int64
	strInt64, _ = strconv.ParseInt(str, 10, 64)
	fmt.Println(strInt64)
	var intStr int
	intStr = 100
	// int > string
	intToString := strconv.Itoa(intStr)
	fmt.Println(intToString)
	// int64 > string
	int64ToStr := strconv.FormatInt(strInt64, 10)
	fmt.Println(int64ToStr)
}

package main

import "fmt"

func main() {
	var int1 int
	int1 = 21
	var int2 int64
	// int > int64
	int2 = int64(int1)
	fmt.Println(int2)
	var inta int64
	inta = 10
	var intb int
	// int64 > int64
	intb = int(inta)
	fmt.Println(intb)
	u := uint32(int1)
	fmt.Println(u)
}
